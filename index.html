　　　)local result,value = VarLib2:getGlobalVarByName(vartype,"name")--biến toàn cầu

　　　)local result,table_ = Valuegroup:getAllGroupItem(vartype,"name", playerid) --biến toàn cầu (playerid = 0) hoặc biến riêng (playerid)
